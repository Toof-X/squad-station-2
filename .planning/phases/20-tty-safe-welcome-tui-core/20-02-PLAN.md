1. In `src/commands/welcome.rs`, add a pure function for testable routing logic:
```rust
/// Determine the action for a given keypress in the welcome TUI.
/// Returns None if the key should be ignored (countdown continues).
pub fn routing_action(key: KeyCode, has_config: bool) -> Option<WelcomeAction> {
    match key {
        KeyCode::Enter => {
            if has_config {
                Some(WelcomeAction::LaunchDashboard)
            } else {
                Some(WelcomeAction::LaunchInit)
            }
        }
        KeyCode::Char('q') | KeyCode::Esc => Some(WelcomeAction::Quit),
        _ => None,
    }
}
```

2. Refactor the event loop inside `run_welcome_tui()` to use `routing_action()` instead of inline match:
   Replace the key handling block:
```rust
if key.kind == KeyEventKind::Press {
    if let Some(act) = routing_action(key.code, has_config) {
        action = match act {
            WelcomeAction::Quit => None,
            other => Some(other),
        };
        break;
    }
}
```
   Note: `Quit` maps to `None` return (no action), while `LaunchInit`/`LaunchDashboard` map to `Some(action)`.

3. In `src/main.rs`, replace the `let _ = action;` placeholder in the None arm with actual routing:
```rust
None => {
    use std::io::IsTerminal;
    use std::path::PathBuf;
    if std::io::stdout().is_terminal() {
        let has_config = std::path::Path::new("squad.yml").exists();
        let action = commands::welcome::run_welcome_tui(has_config).await?;
        match action {
            Some(commands::welcome::WelcomeAction::LaunchInit) => {
                commands::init::run(PathBuf::from("squad.yml"), false).await?;
            }
            Some(commands::welcome::WelcomeAction::LaunchDashboard) => {
                commands::ui::run().await?;
            }

        }
    } else {

    }
    Ok(())
}
```

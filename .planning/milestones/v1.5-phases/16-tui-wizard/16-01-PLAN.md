```rust
    Frame, Terminal,

fn setup_terminal() -> anyhow::Result<Terminal<CrosstermBackend<std::io::Stdout>>> {
    enable_raw_mode()?;
    execute!(std::io::stdout(), EnterAlternateScreen)?;
    Terminal::new(CrosstermBackend::new(std::io::stdout())).map_err(Into::into)
}

fn restore_terminal(terminal: &mut Terminal<CrosstermBackend<std::io::Stdout>>) -> anyhow::Result<()> {
    disable_raw_mode()?;
    execute!(terminal.backend_mut(), LeaveAlternateScreen)?;
    terminal.show_cursor()?;
    Ok(())
}
```

**Public `run()` function implementation:**
```rust
pub async fn run() -> anyhow::Result<Option<WizardResult>> {
    let original_hook = std::panic::take_hook();
    std::panic::set_hook(Box::new(move |info| {
        let _ = disable_raw_mode();
        let _ = execute!(std::io::stdout(), LeaveAlternateScreen);
        original_hook(info);
    }));

    let mut terminal = setup_terminal()?;
    let mut state = WizardState::new();

    loop {
        terminal.draw(|frame| render_page(frame, &state))?;

        if event::poll(std::time::Duration::from_millis(250))? {
            if let event::Event::Key(key) = event::read()? {
                if key.kind == KeyEventKind::Press {
                    // Ctrl+C: cancel wizard
                    if key.code == KeyCode::Char('c') && key.modifiers.contains(KeyModifiers::CONTROL) {
                        restore_terminal(&mut terminal)?;
                        return Ok(None);
                    }

                    match handle_key(&mut state, key.code) {
                        KeyAction::Continue => {}
                        KeyAction::Complete => {
                            restore_terminal(&mut terminal)?;
                            return Ok(Some(state.into_result()));
                        }
                    }
                }
            }
        }
    }
}
```

**`handle_key` function** — dispatches by current page:

For `WizardPage::ProjectName`:
- `Enter`: call `validate_project_name(&state.project_input.value)`. On Ok -> advance to AgentCount. On Err -> set `state.project_input.error = Some(msg)`.
- `Esc`: no-op (first page).
- `Backspace`: `state.project_input.pop()`.
- `Char(c)`: `state.project_input.push(c)`.

For `WizardPage::AgentCount`:
- `Enter`: call `validate_count(&state.count_input.value)`. On Ok(n) -> set `state.agent_count = n`, pre-allocate `state.agents` with `n` empty `AgentDraft` structs (each with `focused_field: AgentField::Role`), advance to `AgentConfig { index: 0 }`. On Err -> set error.
- `Esc`: go back to ProjectName.
- `Backspace`: `state.count_input.pop()`.
- `Char(c)`: `state.count_input.push(c)`.

For `WizardPage::AgentConfig { index }`:
- Behavior depends on `state.agents[index].focused_field`:
  - `AgentField::Role`:
    - `Enter`: validate_role. On Ok -> advance focused_field to Tool. On Err -> set error.
    - `Backspace`: pop from role input.
    - `Char(c)`: push to role input.
  - `AgentField::Tool`:
    - `Enter`: advance focused_field to Model (tool is always valid — enum selector).
    - `Left` / `Right` / `Tab`: cycle tool with `draft.tool = draft.tool.cycle_next()`.
  - `AgentField::Model`:
    - `Enter`: advance focused_field to Description (optional — no validation).
    - `Backspace`: pop from model input.
    - `Char(c)`: push to model input.
  - `AgentField::Description`:
    - `Enter`: if last agent (index == agent_count - 1) -> go to Summary. Else -> go to `AgentConfig { index: index + 1 }` and reset next draft's focused_field to Role.
    - `Backspace`: pop from description input.
    - `Char(c)`: push to description input.
- `Esc`: if focused_field is Role and index == 0 -> go back to AgentCount. If focused_field is Role and index > 0 -> go to `AgentConfig { index: index - 1 }` with focused_field set to Description. If focused_field is not Role -> move focused_field backward (Description->Model->Tool->Role).

For `WizardPage::Summary`:
- `Enter`: return `KeyAction::Complete`.
- `Esc`: go to `AgentConfig { index: agent_count - 1 }` with focused_field set to Description.

**2. Data Fetching (connect-per-refresh):**

CRITICAL: This is the WAL checkpoint starvation prevention strategy.

```rust
async fn fetch_snapshot(db_path: &std::path::Path, selected_agent: Option<&str>) -> anyhow::Result<(Vec<Agent>, Vec<Message>)> {
    let opts = SqliteConnectOptions::new()
        .filename(db_path)
        .read_only(true)
        .journal_mode(SqliteJournalMode::Wal);
    let pool = SqlitePoolOptions::new()
        .max_connections(1)
        .connect_with(opts)
        .await?;

    let agents = db::agents::list_agents(&pool).await?;
    let messages = if let Some(agent) = selected_agent {
        db::messages::list_messages(&pool, Some(agent), None, 50).await?
    } else {
        vec![]
    };
    // pool drops here — WAL reader lock released
    Ok((agents, messages))
}
```

Do NOT reconcile in TUI — TUI is read-only monitoring per CONTEXT.md. Display whatever is in DB as-is.

fn setup_terminal() -> anyhow::Result<Terminal<CrosstermBackend<std::io::Stdout>>> {

fn restore_terminal(terminal: &mut Terminal<CrosstermBackend<std::io::Stdout>>) -> anyhow::Result<()> {
    disable_raw_mode()?;
    execute!(terminal.backend_mut(), LeaveAlternateScreen)?;
    terminal.show_cursor()?;

**5. Event Loop:**

```rust
pub async fn run() -> anyhow::Result<()> {
    let config = config::load_config(std::path::Path::new("squad.yml"))?;
    let db_path = config::resolve_db_path(&config)?;

    // Install panic hook for terminal safety
    let original_hook = std::panic::take_hook();
    std::panic::set_hook(Box::new(move |info| {
        let _ = disable_raw_mode();
        let _ = execute!(std::io::stdout(), LeaveAlternateScreen);
        original_hook(info);
    }));

    let mut terminal = setup_terminal()?;
    let mut app = App::new();
    let refresh_interval = std::time::Duration::from_secs(3);
    let mut last_refresh = tokio::time::Instant::now() - refresh_interval;

    loop {
        // Refresh on interval
        if last_refresh.elapsed() >= refresh_interval {
            let selected = app.selected_agent_name().map(String::from);
            let (agents, messages) = fetch_snapshot(&db_path, selected.as_deref()).await?;
            app.agents = agents;
            app.messages = messages;
            // Auto-select first agent if none selected and agents exist
            if app.agent_list_state.selected().is_none() && !app.agents.is_empty() {
                app.agent_list_state.select(Some(0));
            }
            last_refresh = tokio::time::Instant::now();
        }

        terminal.draw(|f| draw_ui(f, &mut app))?;

        // Poll for key events (250ms timeout for responsive refresh)
        if crossterm::event::poll(std::time::Duration::from_millis(250))? {
            if let crossterm::event::Event::Key(key) = crossterm::event::read()? {
                if key.kind == crossterm::event::KeyEventKind::Press {
                    app.handle_key(key.code);
                }
            }
        }

        if app.quit { break; }
    }

    restore_terminal(&mut terminal)?;
    // Restore default panic hook
    let _ = std::panic::take_hook();
    Ok(())
}
```

Use synchronous `event::poll()` + `event::read()` per RESEARCH.md — do NOT use `EventStream` (requires feature flag not enabled).

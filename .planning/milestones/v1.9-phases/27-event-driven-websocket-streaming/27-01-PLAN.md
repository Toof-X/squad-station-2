   ```rust
           interval.tick().await;
           // Run reconciliation if writable pool available
   ```

   ```rust
   async fn handle_socket(socket: WebSocket, state: AppState) {
       let (mut ws_sender, mut ws_receiver) = socket.split();

       // CRITICAL: Subscribe to broadcast BEFORE building snapshot
       // to avoid missing events during snapshot build (Pitfall 4 from research)
       let mut rx = state.tx.subscribe();

       // Send initial full snapshot
       if let Some(snapshot_json) = build_snapshot(&state).await {
           if ws_sender.send(Message::Text(snapshot_json.into())).await.is_err() {
               return;
           }
       }

       // Forward broadcast events to this WS client
       let mut send_task = tokio::spawn(async move {
           loop {
               match rx.recv().await {
                   Ok(msg) => {
                       if ws_sender.send(Message::Text(msg.into())).await.is_err() {
                           break;
                       }
                   }
                   Err(broadcast::error::RecvError::Lagged(_)) => {
                       // Skip missed messages -- next update has current state (Pitfall 2)
                       continue;
                   }
                   Err(broadcast::error::RecvError::Closed) => break,
               }
           }
       });

       // Listen for client disconnect
       let mut recv_task = tokio::spawn(async move {
           while let Some(Ok(msg)) = ws_receiver.next().await {
               if matches!(msg, Message::Close(_)) {
                   break;
               }
           }
       });

       // Wait for either task to finish, abort the other
       tokio::select! {
           _ = &mut send_task => recv_task.abort(),
           _ = &mut recv_task => send_task.abort(),
       }
   }
   ```

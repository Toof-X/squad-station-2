/// Create a Command for the binary with SQUAD_STATION_DB set
fn cmd_with_db(db_path: &std::path::Path) -> std::process::Command {

    c.env("SQUAD_STATION_DB", db_path.to_str().expect("db path must be valid UTF-8"));

}
